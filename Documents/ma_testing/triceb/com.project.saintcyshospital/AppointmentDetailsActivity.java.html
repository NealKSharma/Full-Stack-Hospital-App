package com.project.saintcyshospital;

import android.os.Bundle;
import android.text.format.DateFormat;
import android.widget.TextView;

import androidx.core.view.ViewCompat;

import com.google.android.material.appbar.MaterialToolbar;
import com.google.android.material.bottomnavigation.BottomNavigationView;

/**
 * Screen that shows the full details for a single appointment.
 * <p>
 * This activity is opened from {@link AppointmentsActivity} when a user taps
 * on an appointment in the list. It expects the appointment data to be passed
 * through the intent extras.
 *  @author quinn beckman
 */
public class AppointmentDetailsActivity extends BaseActivity {

    /**
     * Sets up the layout and fills in the appointment information that was passed
     * in the intent extras.
     *
     * @param b saved instance state, if any
     */
    @Override protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_appointment_details);
        setupBottomNav(0);

        BottomNavigationView bottomNav = findViewById(R.id.bottomNav);

        // Makes sure the bottom nav sits above the system navigation bar.
        ViewCompat.setOnApplyWindowInsetsListener(bottomNav, (view, insets) -> {
            int left = view.getPaddingLeft();
            int top = view.getPaddingTop();
            int right = view.getPaddingRight();

            view.setPadding(left, top, right, 0);
            return insets;
        });

        String name   = getIntent().getStringExtra("patientName");
        String email  = getIntent().getStringExtra("patientEmail");
        String phone  = getIntent().getStringExtra("patientPhone");
        String type   = getIntent().getStringExtra("type");
        String status = getIntent().getStringExtra("status");
        long startMs  = getIntent().getLongExtra("startEpochMillis", 0L);

        ((TextView) findViewById(R.id.tvName)).setText(name);
        ((TextView) findViewById(R.id.tvEmail)).setText(email);
        ((TextView) findViewById(R.id.tvPhone)).setText(phone);
        ((TextView) findViewById(R.id.tvType)).setText(type);
        ((TextView) findViewById(R.id.tvStatus)).setText(status != null ? status.toUpperCase() : "");

        // Formats the date and time.
        String date = DateFormat.format("MMM dd, yyyy", startMs).toString();
        String time = DateFormat.format("h:mm a", startMs).toString();
        ((TextView) findViewById(R.id.tvDate)).setText(date);
        ((TextView) findViewById(R.id.tvTime)).setText(time);
    }
}
