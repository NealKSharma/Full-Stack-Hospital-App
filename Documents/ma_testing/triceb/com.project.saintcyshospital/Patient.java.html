package com.project.saintcyshospital;

import java.io.Serializable;

public class Patient implements Serializable {
    public final long id;
    public final String name;
    public final String mrn;
    public final String dob;
    public final String gender;

    public Patient(long id, String name, String mrn, String dob, String gender) {
        this.id = id;
        this.name = name;
        this.mrn = mrn;
        this.dob = dob;
        this.gender = gender;
    }

    @Override
    public String toString() {
        String left = (name != null && !name.isEmpty()) ? name : "Unknown";
        String right = (mrn != null && !mrn.isEmpty()) ? mrn : "—";
        return left + " — " + right;
    }

}
